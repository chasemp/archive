Basic use of multiprocessing pool.map


import time
from multiprocessing import Pool

def f(x):
    print time.time()
    time.sleep(1)
    print time.time()

def main():
    #four simultaneious workers
    pool = Pool(processes=4)
    print pool.map(f, range(10))

if __name__ == '__main__':
    main()
